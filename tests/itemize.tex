  [currentitem]
  [countdown]
  [counter=\v_strc_itemgroups_counter,
   style=\tfxx,
   width=0.5ExHeight,
   text=,
   maxtext={\phantom{X}},
   distance=4mm]

\definesymbol[visual][\usevisualcounter{currentitem}]
\definevisualcounter
  [fancypage1]

  [counter=userpage,color=blue,style=bold]

\definevisualcounter
  [fancypage2]

  [counter=userpage]

\definevisualcounter
  [fancypage3]

  [counter=userpage,maxwidth=2cm]

  [fancypage4]
